df1 = pd.DataFrame({'F Stat': [f_stat], 'p-value': [p_val]}, index=['Chamber'])

degree_centrality = pd.DataFrame.from_dict(nx.degree_centrality(edges), orient='index', columns=['Degree Centrality'])
